89statusErrDef sendSensorDataToOBDH(const sensorDef sensorId, int32_t sensorValue) {
90 statusErrDef ret = noError;
91
92 // Split sensorId into two bytes
93 uint8_t SensorIdHighByte = (sensorId >> 8) & 0xFF; // High byte of sensorId
94 uint8_t SensorIdLowByte = sensorId & 0xFF; // Low byte of sensorId
95
96 // Split sensorValue (32-bit) into four bytes
97 uint8_t SensorValueByte3 = (sensorValue >> 24) & 0xFF; // Most significant byte
98 uint8_t SensorValueByte2 = (sensorValue >> 16) & 0xFF;
99 uint8_t SensorValueByte1 = (sensorValue >> 8) & 0xFF;
100 uint8_t SensorValueByte0 = sensorValue & 0xFF; // Least significant byte
101
102 // SENSOR_DATA_SIZE is 7 (1 header + 2 for ID + 4 for value)
103 uint8_t data[SENSOR_DATA_SIZE] = {
104 0xFF, // Header byte
105 SensorIdHighByte, // Sensor ID high byte
106 SensorIdLowByte, // Sensor ID low byte
107 SensorValueByte3, // Sensor value most significant byte
108 SensorValueByte2,
109 SensorValueByte1,
110 SensorValueByte0 // Sensor value least significant byte
111 };
112
113 // Debug output
114 /*
115 for(int i = 0; i < 7; i++) {
116 printf(" data[%d] : 0x%02X |",i,data[i]);
117 }
118 printf("\n");
119 */
120
121 struct can_frame frame; // Use classic CAN frame
122 frame.can_id = CAN_ID_OBDH;
123 frame.can_dlc = SENSOR_DATA_SIZE; // Data length code (0-8 bytes)
124 memcpy(frame.data, data, SENSOR_DATA_SIZE);
125
126 // Write the frame (use sizeof(struct can_frame))
127 if (write(socket_can, &frame, sizeof(struct can_frame)) != sizeof(struct can_frame)) {
128 perror("errWriteCANPayload");
129 return errWriteCANPayload;
130 }
131
132 return ret;
133}

189statusErrDef recieveTCFromOBDH() {
190 statusErrDef ret = noError;
191 struct can_frame frame; // Use classic CAN frame
192
193 ssize_t sizeReceived = read(socket_can, &frame, sizeof(struct can_frame));
194 if (sizeReceived > 0) {
195 //constructs an empty instance
196 CCSDSSpacePacket ccsdsPacket;
197 //interpret an input data as a CCSDS SpacePacket
198 try {
199 // Attempt to interpret the packet
200 ccsdsPacket.interpret(frame.data, sizeReceived);
201 } catch (CCSDSSpacePacketException &e) {
202 // Print the exception details to help debug
203 std::cerr << "CCSDS Packet Error: " << e.toString() << std::endl;
204 std::cerr << "Failed to interpret packet of length " << sizeReceived << std::endl;
205 // Optionally, dump the buffer contents for inspection
206 for (size_t i = 0; i < sizeReceived; i++) {
207 std::cout << std::hex << (int)frame.data[i] << " ";
208 }
209 std::cout << std::endl;
210 }
211
212 std::vector<uint8_t> *userData = ccsdsPacket.getUserDataField();
213 mainStateTC = ((*userData)[0] << 8) | (*userData)[1];
214
215 if(mainStateTC == 0x1701) {
216 resetMsgTimer();
217 printf("State has been changed to control mode\n");
218 sendTelemToOBDH(infoStateToPayloadMode);
219 }
220
221 //get APID
222 std::cout << ccsdsPacket.getPrimaryHeader()->getAPIDAsInteger() << std::endl;
223 //dump packet content
224 std::cout << ccsdsPacket.toString() << std::endl;
225 } else {
226 // If there's no data, just continue (EAGAIN or EWOULDBLOCK)
227 if (errno == EAGAIN || errno == EWOULDBLOCK) {
228 // No data available, continue the loop
229 return infoNoDataInCANBuffer;
230 } else {
231 perror("errReadCANTC");
232 return errReadCANTC;
233 }
234 }
235 return ret;

266 double currentTime;

270

274
275 // Check if we've reached or passed the next scheduled send time
