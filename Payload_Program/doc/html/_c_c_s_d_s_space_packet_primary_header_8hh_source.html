97 std::vector<uint8_t> getAsByteVector() {
98 std::vector<uint8_t> result;
99 std::string str = packetVersionNum.to_string() + packetType.to_string() + secondaryHeaderFlag.to_string()
100 + apid.to_string() + sequenceFlag.to_string() + sequenceCount.to_string();
101 size_t length = packetDataLength.to_ulong();
102 std::bitset<32> bits;
103 for (size_t i = 0; i < 32; i++) {
104 if (str[i] == '0') {
105 bits.set(32 - i - 1, 0);
106 } else {
107 bits.set(32 - i - 1, 1);
108 }
109 }
110 result.push_back((uint8_t) (bits >> 24).to_ulong());
111 result.push_back((uint8_t) ((bits << 8) >> 24).to_ulong());
112 result.push_back((uint8_t) ((bits << 16) >> 24).to_ulong());
113 result.push_back((uint8_t) ((bits << 24) >> 24).to_ulong());
114 result.push_back((uint8_t) (length / 0x100));
115 result.push_back((uint8_t) (length % 0x100));
116 return result;
117 }

123 void interpret(const uint8_t* data) {
124 using namespace std;
125 packetVersionNum = bitset<3>((data[0] & 0xe0) >> 5 /* 1110 0000 */);
126 packetType = bitset<1>((data[0] & 0x10) >> 4 /* 0001 0000 */);
127 secondaryHeaderFlag = bitset<1>((data[0] & 0x08) >> 3 /* 0000 1000 */);
128 bitset<3> apid_msb3bits(data[0] & 0x07);
129 bitset<8> apid_lsb8bits(data[1]);
130 for (size_t i = 0; i < 3; i++) {
131 apid.set(i + 8, apid_msb3bits[i]);
132 }
133 for (size_t i = 0; i < 8; i++) {
134 apid.set(i, apid_lsb8bits[i]);
135 }
136 /*
137 cout << "#data[1]=" << (uint32_t) data[1] << endl;
138 cout << "#apid_lsb8bits=" << apid_lsb8bits.to_string() << endl;
139 cout << "#apid =" << apid.to_string() << endl;
140 cout << "#apid=" << (uint32_t) apid.to_ulong() << endl;
141 */
142 sequenceFlag = bitset<2>((data[2] & 0xc0) >> 6 /* 1100 0000 */);
143 bitset<6> sequenceCount_msb6bits(data[2] & 0x3F/* 0011 1111 */);
144 bitset<8> sequenceCount_lsb8bits(data[3]);
145 for (size_t i = 0; i < 6; i++) {
146 sequenceCount.set(i + 8, sequenceCount_msb6bits[i]);
147 }
148 for (size_t i = 0; i < 8; i++) {
149 sequenceCount.set(i, sequenceCount_lsb8bits[i]);
150 }
151 packetDataLength = bitset<16>(data[4] * 0x100 + data[5]);
152 }
