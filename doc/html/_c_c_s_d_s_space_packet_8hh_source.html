156 void interpret(const uint8_t *buffer, size_t length) {
157 using namespace std;
158
159 if (length < 6) {
160 throw CCSDSSpacePacketException(CCSDSSpacePacketException::NotACCSDSSpacePacket);
161 }
162 //primary header
163 primaryHeader->interpret(buffer);
164 size_t packetDataLengthCorrected1 = primaryHeader->getPacketDataLength() + 1;
165 size_t totalPacketLength = packetDataLengthCorrected1 + CCSDSSpacePacketPrimaryHeader::PrimaryHeaderLength;
166
167 if (length < totalPacketLength) {
168 throw CCSDSSpacePacketException(CCSDSSpacePacketException::InconsistentPacketLength);
169 }
170
171 if (primaryHeader->getSecondaryHeaderFlag().to_ulong() == CCSDSSpacePacketSecondaryHeaderFlag::NotPresent) {
172 //buffer field
173 userDataField->clear();
174 for (size_t i = CCSDSSpacePacketPrimaryHeader::PrimaryHeaderLength; i < totalPacketLength; i++) {
175 userDataField->push_back(buffer[i]);
176 }
177 } else {
178 //secondary header
179 secondaryHeader->interpret(buffer + CCSDSSpacePacketPrimaryHeader::PrimaryHeaderLength, //
180 length - CCSDSSpacePacketPrimaryHeader::PrimaryHeaderLength);
181 //buffer field
182 userDataField->clear();
183 for (size_t i = CCSDSSpacePacketPrimaryHeader::PrimaryHeaderLength + secondaryHeader->getLength();
184 i < totalPacketLength; i++) {
185 userDataField->push_back(buffer[i]);
186 }
187 }
