93 if(lineCountSensorParamCSV == -1) {
94 return errOpenParamSensorsFile;
95 }
96 if (lineCountSensorParamCSV <= 1) {
97 printf("No sensors found in file.\n");

99 return noError; // Or return an error if this is invalid
100 }
101
102 lineCountSensorParamCSV--; // Adjust for header
103
104 // Allocate the struct itself
105 paramSensors = (struct paramSensorsStruct*)malloc(sizeof(struct paramSensorsStruct));
106 if (paramSensors == NULL) {
107 perror("Error allocating memory for struct");
108 return errAllocParamSensorStruct;
109 }
110
111 paramSensors->id = (uint16_t*)malloc(lineCountSensorParamCSV * sizeof(uint16_t));
112 paramSensors->minCriticalValue = (int32_t*)malloc(lineCountSensorParamCSV * sizeof(int32_t));
113 paramSensors->minWarnValue = (int32_t*)malloc(lineCountSensorParamCSV * sizeof(int32_t));
114 paramSensors->currentValue = (int32_t*)malloc(lineCountSensorParamCSV * sizeof(int32_t));
115 paramSensors->maxWarnValue = (int32_t*)malloc(lineCountSensorParamCSV * sizeof(int32_t));
116 paramSensors->maxCriticalValue = (int32_t*)malloc(lineCountSensorParamCSV * sizeof(int32_t));
117
118 // Check all allocations
119 if (paramSensors->id == NULL || paramSensors->minCriticalValue == NULL ||
120 paramSensors->minWarnValue == NULL || paramSensors->currentValue == NULL ||
121 paramSensors->maxWarnValue == NULL || paramSensors->maxCriticalValue == NULL) {
122 perror("errAllocParamSensorStruct");
123 // Free any successful allocations to avoid leaks
124 free(paramSensors->id);
125 free(paramSensors->minCriticalValue);
126 free(paramSensors->minWarnValue);
127 free(paramSensors->currentValue);
128 free(paramSensors->maxWarnValue);
129 free(paramSensors->maxCriticalValue);
130 free(paramSensors);
131 paramSensors = NULL;
132 return errAllocParamSensorStruct;
133 }
134
135 // Initialize all arrays to 0
136 memset(paramSensors->id, 0, lineCountSensorParamCSV * sizeof(uint16_t));
137 memset(paramSensors->minCriticalValue, 0, lineCountSensorParamCSV * sizeof(int32_t));
138 memset(paramSensors->minWarnValue, 0, lineCountSensorParamCSV * sizeof(int32_t));
139 memset(paramSensors->currentValue, 0, lineCountSensorParamCSV * sizeof(int32_t));
140 memset(paramSensors->maxWarnValue, 0, lineCountSensorParamCSV * sizeof(int32_t));
141 memset(paramSensors->maxCriticalValue, 0, lineCountSensorParamCSV * sizeof(int32_t));
142
143 ret = readParamSensorsFile(filePath);
144
145 if (paramSensors != NULL) {
146 printf("Number of sensors: %d\n", lineCountSensorParamCSV);
147 for (int i = 0; i < lineCountSensorParamCSV; i++) {
148 printf("Sensor %d: id=0x%04X, minCrit=%d, minWarn=%d, maxWarn=%d, maxCrit=%d\n",
149 i, paramSensors->id[i], paramSensors->minCriticalValue[i],
150 paramSensors->minWarnValue[i], paramSensors->maxWarnValue[i],
151 paramSensors->maxCriticalValue[i]);
152 }
153 } else {
154 printf("paramSensors is NULL\n");
155 }
156
157 return ret;
158}

198statusErrDef readParamSensorsFile(const char* fileName) {
199 FILE* file = fopen(fileName, "r");
200 if (file == NULL)
201 {
202 perror("File open error");
203 return errOpenParamSensorsFile;
204 }
205 int pos = 0;
206
207 char line[MAX_CSV_LINE_SIZE];
208 // Skip the header line
209 if (fgets(line, sizeof(line), file) == NULL) {
210 fclose(file);
211 // File is empty or unreadable
212 return errOpenParamSensorsFile;
213 }
214
215 while (fgets(line, sizeof(line), file) != NULL && pos <= lineCountSensorParamCSV)
216 {
217 line[strcspn(line, "\n")] = 0;
218 fillParamSensorsStruct(line, pos);
219 pos++;
220 }
221
222 fclose(file);
223 return noError;
224}

232void fillParamSensorsStruct(char* line, int pos) {
233 char* token = strtok(line, ";");
234 int column = 0;
235
236 while (token != NULL)
237 {
238 switch (column) {
239 case 1:
240 paramSensors->id[pos] = (uint16_t)strtol(token, NULL, 0);
241 break;
242 case 2:
243 paramSensors->minCriticalValue[pos] = atoi(token);
244 break;
245 case 3:
246 paramSensors->minWarnValue[pos] = atoi(token);
247 break;
248 case 5:
249 paramSensors->maxWarnValue[pos] = atoi(token);
250 break;
251 case 6:
252 paramSensors->maxCriticalValue[pos] = atoi(token);
253 break;
254 default:
255 break;
256 }
257 column++;
258 token = strtok(NULL, ";");
259 }
260}

329statusErrDef initCANSocket() {

334#if USE_VCAN

343 printf("CAN Sockets init\r\n");
344
345 if ((socket_can = socket(PF_CAN, SOCK_RAW, CAN_RAW)) < 0) {
346 perror("errCreateCANSocket");
347 return errCreateCANSocket;
348 }
349
350 int buf_size = CAN_SOCKET_BUFFER_SIZE;
351 if(setsockopt(socket_can, SOL_SOCKET, SO_RCVBUF, &buf_size, sizeof(buf_size)) == -1){
352 perror("errSetCANSocketBufSize");
353 return errSetCANSocketBufSize;
354 }
355
356 int flags = fcntl(socket_can, F_GETFL, 0);
357 if(flags == -1) {
358 perror("errGetCANSocketFlags");
359 return errGetCANSocketFlags;
360 }
361 if(fcntl(socket_can, F_SETFL, flags | O_NONBLOCK) == -1) {
362 perror("errSetCANSocketNonBlocking");
363 return errSetCANSocketNonBlocking;
364 }
365
366 strcpy(ifr.ifr_name, CAN_INTERFACE);
367 ioctl(socket_can, SIOCGIFINDEX, &ifr);
368
369 memset(&addr, 0, sizeof(addr));
370 addr.can_family = AF_CAN;
371 addr.can_ifindex = ifr.ifr_ifindex;
372
373 if (bind(socket_can, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
374 perror("errBindCANAddr");
375 return errBindCANAddr;
376 }
377
378 struct can_filter rfilter[1];
379 rfilter[0].can_id = CAN_ID_OBDH;
380 rfilter[0].can_mask = CAN_SFF_MASK;
381 setsockopt(socket_can, SOL_CAN_RAW, CAN_RAW_FILTER, &rfilter, sizeof(rfilter));
382
383 return ret;

400
401 // Create UDP socket
402 socket_udp = socket(AF_INET, SOCK_DGRAM, 0);
403 if (socket_udp < 0) {
404 perror("Socket creation failed");
405 return errCreateUDPSocket;
406 }
407
408 int buf_size = UDP_SOCKET_BUFFER_SIZE;
409 if(setsockopt(socket_udp, SOL_SOCKET, SO_RCVBUF, &buf_size, sizeof(buf_size)) == -1){
410 perror("errSetUDPSocketBufSize");
411 return errSetUDPSocketBufSize;
412 }
413
414 int flags = fcntl(socket_udp, F_GETFL, 0);
415 if(flags == -1) {
416 perror("errGetUDPSocketFlags");
417 return errGetUDPSocketFlags;
418 }
419 if(fcntl(socket_udp, F_SETFL, flags | O_NONBLOCK) == -1) {
420 perror("errSetUDPSocketNonBlocking");
421 return errSetUDPSocketNonBlocking;
422 }
423
424 // Bind to UDP port
425 memset(&serverAddr, 0, sizeof(serverAddr));
426 serverAddr.sin_family = AF_INET;
427 serverAddr.sin_addr.s_addr = INADDR_ANY;
428 serverAddr.sin_port = htons(UDP_TELECOMMAND_PORT);
429
430 if (bind(socket_udp, (struct sockaddr*)&serverAddr, sizeof(serverAddr)) < 0) {
431 perror("Bind failed");
432 return errBindUDPAddr;
433 }
434
435
436 printf("Listening for cFS packets on UDP port %d ...\n", UDP_TELECOMMAND_PORT);
