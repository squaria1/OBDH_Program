130public:
136 ADU* unite() {
137 if (!complete) {
138 throw ADUSegmentsException();
139 }
140 ADU* adu = new ADU;
141 adu->category = pendingPackets[0]->getSecondaryHeader()->getCategory().to_ulong();
142 adu->ADUChannelID = this->aduChannelID;
143 adu->packettype = pendingPackets[0]->getPrimaryHeader()->getPacketType().to_ulong();
144 adu->upperAPID = pendingPackets[0]->getPrimaryHeader()->getAPIDAsInteger() >> 8;
145 adu->lowerAPID = pendingPackets[0]->getPrimaryHeader()->getAPIDAsInteger() & 0xff;
146 adu->ADUCount = pendingPackets[0]->getSecondaryHeader()->getADUCount();
147 adu->TI = pendingPackets[0]->getSecondaryHeader()->getTimeAsInteger();
148 for (size_t i = 0; i < pendingPackets.size(); i++) {
149 std::vector<uint8_t>* nextdata = pendingPackets[i]->getUserDataField();
150 adu->data.insert(adu->data.end(), nextdata->begin(), nextdata->end());
151 }
152 initialize();
153 return adu;
154 }

156private:
157 void error_process(std::string message = "") {
158 initialize();
159 currentSegmentFlag = ErrorSegment;
160 throw ADUSegmentsException(message);
161 }

163public:
164 static const size_t NMaximumADUSegmentCount = 16384;
165
166public:
167 void push(CCSDSSpacePacket* packet) {
168 if (packet->isTCPacket()) {
169 using namespace std;
170 cerr << "ADUUnsegmenter::push(): Warining. TC Packet was pushed." << endl;
171 return;
172 }
173 CCSDSSpacePacketSecondaryHeader* secondaryHeader = packet->getSecondaryHeader();
174 if (complete) {
175 throw ADUSegmentsException();
176 } else if (secondaryHeader->getADUChannelID() != this->aduChannelID) {
177 throw ADUSegmentsException();
178 }
179
180 if (pendingPackets.size() != 0) {
181 if (secondaryHeader->getADUCount() != pendingPackets[0]->getSecondaryHeader()->getADUCount()) {
182 using namespace std;
183 std::stringstream ss;
184 ss << "ADUSegments::push(): Different ADU Count: "
185 << dec << (int) secondaryHeader->getADUCount() << "<-->" << (int) pendingPackets[0]->getSecondaryHeader()->getADUCount()
186 << " for ADU Channel ID " << "0x" << hex << right << setw(2) << setfill('0')
187 << (uint32_t) secondaryHeader->getADUChannelID() << endl;
188 ss << "ADUSegments::push(): ADU Segment Counter for ADU Channel ID " << hex << right << setw(2)
189 << setfill('0') << (uint32_t) secondaryHeader->getADUChannelID() << " will be reset to 0." << endl;
190 error_process(ss.str());
191 }
192 }
193
194 uint16_t newSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
195 std::stringstream ss;
196 ss << (int) secondaryHeader->getADUChannelID() << " " << secondaryHeader->getADUSegmentFlag().to_ulong() << " "
197 << currentSegmentCount << " " << newSegmentCount;
198 if (watchADUSegmentCount) {
199 if (currentSegmentFlag != EmptySegment
200 && secondaryHeader->getADUSegmentFlag().to_ulong() != 0x01 /* not First Segment */) {
201 if (newSegmentCount != currentSegmentCount + 1 //i.e. newSegmentCount is different from currentSegmentCount by more than 1
202 && ((size_t) newSegmentCount + NMaximumADUSegmentCount) != ((size_t) currentSegmentCount + 1) //
203 ) {
204 using namespace std;
205 std::stringstream ss;
206 ss << "ADUSegments::push(): ADU Segment Count jumped: " << dec << currentSegmentCount << "-->" << newSegmentCount
207 << " for ADU Channel ID " << "0x" << hex << right << setw(2) << setfill('0')
208 << (uint32_t) secondaryHeader->getADUChannelID() << endl;
209 ss << "ADUSegments::push(): ADU Segment Counter for ADU Channel ID " << hex << right << setw(2)
210 << setfill('0') << (uint32_t) secondaryHeader->getADUChannelID() << " will be reset to 0." << endl;
211 error_process(ss.str());
212 }
213 }
214 }
215
216 if (secondaryHeader->isADUUnsegmented()) {
217 if (currentSegmentFlag == EmptySegment || currentSegmentFlag == ErrorSegment) {
218 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
219 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
220 clearAndDeletePendingPackets();
221 pendingPackets.push_back(packet);
222 complete = true;
223 } else {
224 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
225 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
226 error_process("uns");
227 }
228 } else if (secondaryHeader->isADUFirstSegment()) {
229 if (currentSegmentFlag == EmptySegment || currentSegmentFlag == ErrorSegment) {
230 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
231 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
232 clearAndDeletePendingPackets();
233 pendingPackets.push_back(packet);
234 complete = false;
235 } else {
236 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
237 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
238 error_process("1st");
239 }
240 } else if (secondaryHeader->isADUContinuationSegment()) {
241 if (currentSegmentFlag == CCSDSSpacePacketSequenceFlag::TheFirstSegment
242 || currentSegmentFlag == CCSDSSpacePacketSequenceFlag::ContinuationSegment) {
243 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
244 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
245 pendingPackets.push_back(packet);
246 complete = false;
247 } else {
248 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
249 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
250 error_process("tbc");
251 }
252 } else if (secondaryHeader->isADULastSegment()) {
253 if (currentSegmentFlag == CCSDSSpacePacketSequenceFlag::TheFirstSegment
254 || currentSegmentFlag == CCSDSSpacePacketSequenceFlag::ContinuationSegment) {
255 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
256 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
257 pendingPackets.push_back(packet);
258 complete = true;
259 } else {
260 currentSegmentFlag = secondaryHeader->getADUSegmentFlag().to_ulong();
261 currentSegmentCount = secondaryHeader->getADUSegmentCount().to_ulong();
262 error_process("fin");
263 }
264 } else {
265 error_process("hatena");
266 }
267 }

321public:
332 void push(const std::vector<uint8_t>& ccsdsSpacePacketByteArray) throw (ADUUnsegmenterException,
333 CCSDSSpacePacketException) {
334 CCSDSSpacePacket* packet = new CCSDSSpacePacket;
335 packet->interpret(&ccsdsSpacePacketByteArray[0], ccsdsSpacePacketByteArray.size());
336 if (packet->isTCPacket()) {
337 delete packet;
338 return;
339 }
340 CCSDSSpacePacketPrimaryHeader* primaryHeader = packet->getPrimaryHeader();
341 CCSDSSpacePacketSecondaryHeader* secondaryHeader = packet->getSecondaryHeader();
342 uint16_t lowerAPID = primaryHeader->getAPIDAsInteger() & 0xff;
343 if (lowerAPID != this->lowerAPID) {
344 delete packet;
345 ADUUnsegmenterException e("APIDMismatch");
346 throw e;
347 }
348
349 uint16_t aduChannelID = secondaryHeader->getADUChannelID();
350 if (aduSegmentMap.find(aduChannelID) == aduSegmentMap.end()) {
351 aduSegmentMap[aduChannelID] = new ADUSegments(aduChannelID);
352 aduSegmentMap[aduChannelID]->watchADUSegmentCount = true;
353 }
354 ADUSegments* segments = aduSegmentMap[aduChannelID];
355 try {
356 segments->push(packet);
357 } catch (ADUSegmentsException& e) {
358 throw ADUUnsegmenterException(e.toString());
359 }
360
361 if (segments->isComplete()) {
362 ADU* adu = segments->unite();
363 completedADUs.push(adu);
364 }
365 }
